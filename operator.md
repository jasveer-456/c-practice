# operator
 ## input
 ```
#include <stdio.h>

void main(){
	int x=10;
	int y=20;
	int z=x+y;
	printf("%d",z); 
	int a=40;
	int b=80;
	int c=b-a;
	printf("\n%d",c);
	int d =45;
	int e =45;
	int f =d*e;
	printf("\n%d",f);
	int g = 48;
	int h = 2;
	int i = g/h;
	printf("\n%d",i);
	int j = 10;
	int k = 20;
	int l =j%k;
	printf("\n%d",l);
	int m =30;
	m = ++m;
	printf("\n%d",m);
	int n = 40;
	n =--n;
	printf("\n%d",n);
}
```
